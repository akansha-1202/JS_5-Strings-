QUESTION 2.  Count Characters

You are given a string S, and your task is to return an array B (having a size of 2), where B [0] contains the count of character A (uppercase) in string S and B[1] contains the count of character D (uppercase) in string S.

SOLUTION. 
      const countCharacters = (S) => 
      {
        let countOfA=0, countOfD=0;
        for(let i=0;i < S.length;i++){
          if(S[i]=="A")
             countOfA++;
          if(S[i]=="D")
          countOfD++;
        }

       //array declaration for storing values:countOfA,countOfD

       let result=[];
       result.push(countOfA);
       result.push(countOfD);
       return result;
      };
Input: AbaDd
Output: 11

SOLUTION. 
var Game_Winner = (S) => 
        {
          let countOfA=0;
          let countOfD=0;
          for(let i=0;i< S.length;i++)
          {
            if(S[i]==="A")
            {
             countOfA++;
            }
            else
             {
               countOfD++;

        }

QUESTION 13.  Count the Vowels and Consonants

You are given a string S containing both uppercase and lowercase letters. You need to return out the number of vowels and the number of consonants in the given string.

SOLUTION. 
          var Count_Vowels= (S) => 
          {
             let countOfVowels=0;
             for(let i=0;i< S.length;i++)
             {
               if(S[i]==="A" || S[i]==="a" ||
                  S[i]==="E" || S[i]==="e" ||
                  S[i]==="I" || S[i]==="i" ||
                  S[i]==="O" || S[i]==="o" ||
                  S[i]==="U" || S[i]==="u"
                  )
                {
                  countOfVowels++;
                }
             }
             return countOfVowels;
          };
          var Count_Consonants= (S) => 
          {
            let countOfConsonants=0;
             for(let i=0;i< S.length;i++)
             {
               if(S[i]!=="A" && S[i]!=="a" &&
                  S[i]!=="E" && S[i]!=="e" &&
                  S[i]!=="I" && S[i]!=="i" &&
                  S[i]!=="O" && S[i]!=="o" &&
                  S[i]!=="U" && S[i]!=="u" 
                  )
                {
                  countOfConsonants++;
                }
             }
             return countOfConsonants;     
          };
